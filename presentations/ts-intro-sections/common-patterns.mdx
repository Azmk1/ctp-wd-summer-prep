## Real World Example
<CodeDisplay height="500px">
  <File name="example.typescript">
    
```typescript
interface TodoItem {
  id: string;
  title: string;
  completed: boolean;
  createdAt: Date;
}

type TodoFilter = "all" | "active" | "completed";

function filterTodos(todos: TodoItem[], filter: TodoFilter): TodoItem[] {
  switch (filter) {
    case "active":
      return todos.filter(todo => !todo.completed);
    case "completed":
      return todos.filter(todo => todo.completed);
    default:
      return todos;
  }
}
```
