## Migrating from JavaScript to TypeScript

- Start gradually, not all at once
- Use `.ts` files alongside `.js` files
- Enable `allowJs` in tsconfig.json
- Convert file by file

```json
// tsconfig.json for migration
{
  "compilerOptions": {
    "allowJs": true,
    "checkJs": true,
    "strict": false, // Start loose, tighten over time
    "noImplicitAny": false,
    "strictNullChecks": false
  }
}
```
</SlideLayout.Full>

## Migration Steps

1. **Setup TypeScript** - Install and configure
2. **Rename files** - `.js` → `.ts` (or `.jsx` → `.tsx`)
3. **Add basic types** - Start with function parameters
4. **Fix errors** - Address type errors one by one
5. **Improve types** - Refine and add more specific types
6. **Enable strict mode** - Gradually enable strict checks

```typescript
// Before (JavaScript)
function calculateTotal(items) {
  return items.reduce((sum, item) => sum + item.price, 0);
}

// After (TypeScript)
interface Item {
  name: string;
  price: number;
}

function calculateTotal(items: Item[]): number {
  return items.reduce((sum, item) => sum + item.price, 0);
}
```

## Common Migration Patterns

// Pattern 1: Start with JSDoc
/**
 * @param {string} name
 * @param {number} age
 * @returns {string}

function greet(name, age) {
  return `Hello ${name}, you are ${age} years old`;

// Pattern 2: Gradual typing
function processData(data: any): any { // Start with any

// Later, refine the types
interface InputData {
  id: string;
  values: number[];
}

interface OutputData {
  id: string;
  result: number;
}

function processData(data: InputData): OutputData {

## TypeScript Strict Mode
<CodeDisplay >
  <File name="example.typescript">
    
```typescript
// Start with loose config
{
  "compilerOptions": {
    "strict": false
  }
}

// Gradually enable strict checks
{
  "compilerOptions": {
    "noImplicitAny": true,      // Phase 1
    "strictNullChecks": true,   // Phase 2
    "strictFunctionTypes": true, // Phase 3
    "strict": true              // Final: All strict checks
  }
}

// Fix issues as you enable each option
// Before: function might return undefined
function findUser(id) {
  return users.find(u => u.id === id);
}

// After: Explicit about possible undefined
function findUser(id: string): User | undefined {
  return users.find(u => u.id === id);
}
```
