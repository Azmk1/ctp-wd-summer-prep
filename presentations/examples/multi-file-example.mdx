interface Habit {
    id: string;
    name: string;
    streak: number;
}

const HabitCard: React.FC<{ habit: Habit }> = ({ habit }) => {
    return (
        <div className="habit-card">
            <h3 className="habit-title">{habit.name}</h3>
            <div className="habit-streak">Streak: {habit.streak} days</div>
        </div>
    );
};

const App: React.FC = () => {
    const habits: Habit[] = [
        { id: '1', name: 'Morning Exercise', streak: 15 },
        { id: '2', name: 'Read 30 minutes', streak: 7 }
    ];
    
    return (
        <div>
            <h1>My Habits</h1>
            {habits.map(habit => (
                <HabitCard key={habit.id} habit={habit} />
            ))}
        </div>
    );
};

    const cardRule = cssRules.find(r => r.selector === '.habit-card');

    expect(cardRule.declarations['border-radius']).to.equal('8px');
});

test('Should display habit names', () => {
    const titles = document.querySelectorAll('.habit-title');
    expect(titles[0].textContent).to.equal('Morning Exercise');
