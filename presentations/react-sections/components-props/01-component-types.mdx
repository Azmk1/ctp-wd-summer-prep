import React from 'react';

// Function Component (Recommended)
function HabitStreak(props: { days: number }): JSX.Element {

const HabitProgress: React.FC<{ completed: number; total: number }> = ({ completed, total }) => {
  const percentage = Math.round((completed / total) * 100);
  return <div className="progress">Progress: {percentage}%</div>;
};

// Class Component (Legacy - still supported)
