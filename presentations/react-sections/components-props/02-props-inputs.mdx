import React from 'react';
import ReactDOM from 'react-dom/client';

// Define prop types with TypeScript interface
interface HabitCardProps {
  name: string;
  frequency: 'daily' | 'weekly' | 'monthly';
  completedToday: boolean;
  currentStreak: number;
  onToggle?: () => void; // Optional prop
}

// Component with typed props
function HabitCard({ 
  name, 
  frequency, 
  completedToday, 
  currentStreak,
  onToggle 
}: HabitCardProps): JSX.Element {
  return (
    <div className={`habit-card ${completedToday ? 'completed' : ''}`}>
      <h3>{name}</h3>
      <p className="frequency">Goal: {frequency}</p>
      <p className="streak">Current streak: {currentStreak} days</p>
      <button 
        onClick={onToggle}
        className={`toggle-btn ${completedToday ? 'checked' : ''}`}
      >
        {completedToday ? 'âœ“ Completed' : 'Mark Complete'}
      </button>
    </div>
  );
}

// Props can be any type
interface HabitStatsProps {
  totalHabits: number;
  completedToday: number;
  bestStreak: number;
  completionRate: number;
}

function HabitStats({ 
  totalHabits, 
  completedToday, 
  bestStreak,
  completionRate 
}: HabitStatsProps): JSX.Element {
  return (
    <div className="habit-stats">
      <div className="stat">
        <span className="stat-value">{completedToday}/{totalHabits}</span>
        <span className="stat-label">Today</span>
      </div>
      <div className="stat">
        <span className="stat-value">{bestStreak}</span>
        <span className="stat-label">Best Streak</span>
      </div>
      <div className="stat">
        <span className="stat-value">{completionRate}%</span>
        <span className="stat-label">Success Rate</span>
      </div>
    </div>
  );
}

function App(): JSX.Element {
  const handleToggle = () => alert('Habit toggled!');

  return (
    <div className="app">
      <HabitStats 
        totalHabits={5} 
        completedToday={3} 
        bestStreak={21}
        completionRate={85}
      />
      <HabitCard 
        name="Drink 8 glasses of water" 
        frequency="daily" 
        completedToday={true}
        currentStreak={7}
        onToggle={handleToggle}
      />
      <HabitCard 
        name="Exercise for 30 minutes" 
        frequency="daily" 
        completedToday={false}
        currentStreak={3}
        onToggle={handleToggle}
      />
    </div>
  );
}

// Render the app
const root = ReactDOM.createRoot(document.getElementById('root')!);
root.render(<App />);
