import React from 'react';
// Plain CSS import
import './styles.css';
// CSS Module import
import styles from './Button.module.css';

// Using plain CSS
export function PlainCSSButton({ 
  variant = 'primary', 
  children, 
  disabled 
}: ButtonProps) {
  const className = `button button-${variant}`;
  
  return (
    <button className={className} disabled={disabled}>
      {children}
    </button>
  );
}

// Using CSS Modules
export function CSSModuleButton({ 
  variant = 'primary', 
  size,
  icon,
  children, 
  disabled 
}: ButtonProps) {
  const classNames = [
    styles[variant],
    size && styles[size],
    icon && styles.icon
  ].filter(Boolean).join(' ');
  
  return (
    <button className={classNames} disabled={disabled}>
      {icon && <span>{icon}</span>}
      {children}
    </button>
  );
}

// Using clsx for conditional classes
import clsx from 'clsx';

export function ButtonWithClsx({ 
  variant = 'primary',
  size,
  fullWidth,
  children 
}: ButtonProps) {
  return (
    <button 
      className={clsx(
        styles.button,
        styles[variant],
        {
          [styles.large]: size === 'large',
          [styles.fullWidth]: fullWidth
        }
      )}
    >
      {children}
    </button>
  );
}

interface ButtonProps {
  variant?: 'primary' | 'secondary';
  size?: 'small' | 'medium' | 'large';
  icon?: React.ReactNode;
  fullWidth?: boolean;
  children: React.ReactNode;
  disabled?: boolean;
}

// Example usage
function App() {
  return (
    <div>
      <PlainCSSButton>Plain CSS</PlainCSSButton>
      <CSSModuleButton variant="secondary" icon="ðŸš€">
        CSS Module
      </CSSModuleButton>
      <ButtonWithClsx variant="primary" size="large" fullWidth>
        Full Width Large
      </ButtonWithClsx>
    </div>
  );
}

export default App;
