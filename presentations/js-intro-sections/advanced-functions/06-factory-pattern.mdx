## Closure Benefits: Factory Pattern
```javascript
function createHabitTracker(name) {
  let completions = 0;
  let streak = 0;
  return {
    complete() {
      completions++;
      streak++;
      return `${name}: ${streak} day streak!`;
    },
    reset() {
      streak = 0;
    },
    getStats() {
      return { completions, streak };
    }
  };
}
```
