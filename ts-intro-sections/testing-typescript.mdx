// Jest with TypeScript
import { Calculator } from './calculator';

describe('Calculator', () => {
  let calculator: Calculator;
  
  beforeEach(() => {
    calculator = new Calculator();
  });
  
  it('should add numbers correctly', () => {
    expect(calculator.add(2, 3)).toBe(5);
    expect(calculator.add(-1, 1)).toBe(0);
  });
  
  it('should handle division by zero', () => {
    expect(() => calculator.divide(10, 0)).toThrow('Division by zero');
  });
});

## Type-Safe Mocking

```typescript
// Mocking with proper types
interface UserService {
  getUser(id: string): Promise<User>;
  saveUser(user: User): Promise<void>;
}

// Create a mock that satisfies the interface
const mockUserService: jest.Mocked<UserService> = {
  getUser: jest.fn(),
  saveUser: jest.fn()
};

// Type-safe mock implementations
mockUserService.getUser.mockResolvedValue({
  id: '123',
  name: 'Test User',
  email: 'test@example.com'
});

// Using the mock in tests
test('should fetch user', async () => {
  const user = await mockUserService.getUser('123');
  expect(user.name).toBe('Test User');
  expect(mockUserService.getUser).toHaveBeenCalledWith('123');
});
```

## Testing Utilities

```typescript
// Custom type-safe test utilities
function createMockUser(overrides?: Partial<User>): User {
  return {
    id: '1',
    name: 'Test User',
    email: 'test@example.com',
    createdAt: new Date(),
    ...overrides
  };
}

// Type-safe test data builders
class UserBuilder {
  private user: Partial<User> = {};
  
  withName(name: string): this {
    this.user.name = name;
    return this;
  }
  
  withEmail(email: string): this {
    this.user.email = email;
    return this;
  }
  
  build(): User {
    return {
      id: '1',
      name: 'Default',
      email: 'default@example.com',
      createdAt: new Date(),
      ...this.user
    };
  }
}

// Usage
const testUser = new UserBuilder()
  .withName('Alice')
  .withEmail('alice@example.com')
  .build();
```

## Testing React Components

```typescript
// Testing React components with TypeScript
import { render, fireEvent, screen } from '@testing-library/react';
import { Button, ButtonProps } from './Button';

const defaultProps: ButtonProps = {
  text: 'Click me',
  onClick: jest.fn(),
  variant: 'primary'
};

function renderButton(props: Partial<ButtonProps> = {}) {
  return render(<Button {...defaultProps} {...props} />);
}

describe('Button', () => {
  it('should render with text', () => {
    renderButton();
    expect(screen.getByText('Click me')).toBeInTheDocument();
  });
  
  it('should call onClick when clicked', () => {
    const onClick = jest.fn();
    renderButton({ onClick });
    
    fireEvent.click(screen.getByText('Click me'));
    expect(onClick).toHaveBeenCalledTimes(1);
  });
  
  it('should be disabled when disabled prop is true', () => {
    renderButton({ disabled: true });
    expect(screen.getByText('Click me')).toBeDisabled();
  });
});
```
